import java.awt.*; import java.util.Timer; import java.util.TimerTask; public class MouseLocker { public static void main(String[] args) { try { Robot robot = new Robot(); // Get screen size Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize(); int centerX = screenSize.width / 2; int centerY = screenSize.height / 2; // Move mouse to center repeatedly Timer timer = new Timer(); timer.scheduleAtFixedRate(new TimerTask() { @Override public void run() { robot.mouseMove(centerX, centerY); } }, 0, 10); // every 10 milliseconds } catch (AWTException e) { e.printStackTrace(); } } }
